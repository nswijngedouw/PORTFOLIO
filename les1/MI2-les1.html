Voorbeeld 8 : functie als deel van een prototype

function Rekenmachine(){
	this.waarde1 =0;
	this.waarde2 =0;
							
	this.Optellen = function(){
		return this.waarde1 + this.waarde2;
	}
}
var rekenen = new Rekenmachine();
rekenen.waarde1 = 5;
rekenen.waarde2 = 10;
var uitkomst = rekenen.Optellen();
	
document.write("uitkomst : " + uitkomst);
