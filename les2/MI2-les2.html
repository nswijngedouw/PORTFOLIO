Voorbeeld 7 : functiedeclaratie vs functie expressie

/*
 * Deze instructie lukt niet doordat 'optellen' op dit moment
 * de functie 'optellen' nog niet herkend wordt (zie keyword
 * voor optellen).
 */
var uitkomst1 = optellen(10,20);
/*
 * Deze instructie wordt wel herkend
 */
var uitkomst2 = vermenigvuldigen(10,20);

var optellen = function(x,y){
	return x + y;
}	

function vermenigvuldigen(x,y){
	return x * y;
}

document.write("uitkomst 1 : " + uitkomst1+'<br>');
document.write("uitkomst 2 : " + uitkomst2+'<br>');
