Achtergrondkleur opvragen met javascript

// Met IE 6 & 7
var headerElement = document.getElementById("resultaat1");
var headerBackground = headerElement.currentStyle.backgroundColor;	

// Met FF 3+
var headerElement = document.getElementById("resultaat1");
var headerStyle = document.defaultView.getComputedStyle(headerElement, null);
var headerBackground = headerStyle.backgroundColor;
